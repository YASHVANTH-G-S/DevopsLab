package org.apache.maven.achetypes.maven_achetype_quickstart;



import java.io.FileReader;
import java.io.IOException;

import org.json.simple.JSONObject;
import org.json.simple.parser.JSONParser;
import org.json.simple.parser.ParseException;

public class ReadJSON {

    public static void main(String[] args) throws IOException, ParseException {
        
        // JSON parser object to parse the JSON file
        JSONParser jsonParser = new JSONParser();
        
        // Location of JSON file
        FileReader reader = new FileReader(".\\JSON\\student.json");
        
        // Parse JSON file
        Object obj = jsonParser.parse(reader);
        JSONObject studentObj = (JSONObject) obj;
        
        // Fetch values from JSON using keys
        String fname = (String) studentObj.get("firstname");
        String lname = (String) studentObj.get("lastname");
        
        // Print values on console
        System.out.println("Firstname: " + fname);
        System.out.println("Lastname: " + lname);
    }
}
